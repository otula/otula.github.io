016package service.tut.pori.fuzzysensors.frontend;
017
018import java.util.UUID;
019
020import org.apache.log4j.Logger;
021import org.springframework.util.ReflectionUtils;
022
023import service.tut.pori.apilta.sensors.datatypes.SensorTask;
024import service.tut.pori.apilta.sensors.reference.SensorsReferenceCore;
025import service.tut.pori.fuzzysensors.FuzzyTask;
026import core.tut.pori.http.parameters.DataGroups;
027import core.tut.pori.http.parameters.Limits;
028import core.tut.pori.users.UserIdentity;
029
030/**
031 * fuzzy core methods for the fuzzy front end
032 *
033 */
034public final class FuzzyFrontendCore {
035  private static final Logger LOGGER = Logger.getLogger(FuzzyFrontendCore.class);
036  
037  /**
038   * 
039   */
040  private FuzzyFrontendCore() {
041    // nothing needed
042  }
043
044  /**
045   * 
046   * @param authenticatedUser 
047   * @param backendId
048   * @param taskId
049   * @param dataGroups
050   * @param limits
051   * @return fuzzy task
052   */
053  public static FuzzyTask queryTaskDetails(UserIdentity authenticatedUser, Long backendId, String taskId, DataGroups dataGroups, Limits limits) {
054    SensorTask sTask = SensorsReferenceCore.generateTaskDetails(authenticatedUser, backendId, dataGroups, limits, service.tut.pori.apilta.sensors.Definitions.TASK_TYPE_DATA_COLLECT); // create a simple sensor task
055    FuzzyTask fTask = new FuzzyTask();
056    ReflectionUtils.shallowCopyFieldState(sTask, fTask); //TODO check that this really works!!!
057    return fTask;
058  }
059
060  /**
061   * 
062   * @param authenticatedUser 
063   * @param task
064   * @return id for the generated task
065   */
066  public static String createTask(UserIdentity authenticatedUser, FuzzyTask task) {
067    if(UserIdentity.isValid(authenticatedUser)){
068      LOGGER.debug("Authenticated user, id: "+authenticatedUser.getUserId()); // simply log the user id for debug
069    }
070    return UUID.randomUUID().toString();
071  }
072}
