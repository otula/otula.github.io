016package service.tut.pori.fuzzysensors.frontend;
017
018import core.tut.pori.http.Response;
019import core.tut.pori.http.annotations.HTTPAuthenticationParameter;
020import core.tut.pori.http.annotations.HTTPMethodParameter;
021import core.tut.pori.http.annotations.HTTPService;
022import core.tut.pori.http.annotations.HTTPServiceMethod;
023import core.tut.pori.http.parameters.AuthenticationParameter;
024import core.tut.pori.http.parameters.DataGroups;
025import core.tut.pori.http.parameters.InputStreamParameter;
026import core.tut.pori.http.parameters.Limits;
027import core.tut.pori.http.parameters.LongParameter;
028import core.tut.pori.http.parameters.StringParameter;
029import core.tut.pori.utils.XMLFormatter;
030import service.tut.pori.apilta.sensors.datatypes.SensorTask;
031import service.tut.pori.apilta.sensors.reference.SensorsReferenceCore;
032import service.tut.pori.fuzzysensors.Definitions;
033import service.tut.pori.fuzzysensors.FuzzyTask;
034import service.tut.pori.tasks.datatypes.Task;
035import service.tut.pori.tasks.datatypes.TaskList;
036
037/**
038 * fuzzy declarations for the fuzzy front end service
039 *
040 */
041@HTTPService(name=Definitions.SERVICE_FUZZY_SENSORS_FRONTEND)
042public class FuzzyFrontendService {
043  private XMLFormatter _formatter = new XMLFormatter();
044  
045  /**
046   * 
047   * @param authenticatedUser 
048   * @param xml Only the result data should be in the body. See {@link service.tut.pori.fuzzysensors.FuzzyTask}
049   */
050  @HTTPServiceMethod(name = service.tut.pori.tasks.Definitions.METHOD_TASK_FINISHED, acceptedMethods={core.tut.pori.http.Definitions.METHOD_POST})
051  public void taskFinished (
052      @HTTPAuthenticationParameter AuthenticationParameter authenticatedUser, // require authentication, but accept any known user
053      @HTTPMethodParameter(name = InputStreamParameter.PARAMETER_DEFAULT_NAME, bodyParameter = true) InputStreamParameter xml
054      ) 
055  {
056    SensorsReferenceCore.taskFinished(authenticatedUser.getUserIdentity(), _formatter.toObject(xml.getValue(), FuzzyTask.class)); // simply dump it into the default reference implementation, we do not process the task finished anyway
057  }
058  
059  /**
060   * 
061   * @param authenticatedUser 
062   * @param taskId
063   * @param backendId
064   * @param dataGroups
065   * @param limits paging limits
066   * @return response See {@link service.tut.pori.fuzzysensors.FuzzyTask}
067   */
068  @HTTPServiceMethod(name = service.tut.pori.tasks.Definitions.METHOD_QUERY_TASK_DETAILS, acceptedMethods={core.tut.pori.http.Definitions.METHOD_GET})
069  public Response queryTaskDetails (
070      @HTTPAuthenticationParameter AuthenticationParameter authenticatedUser, // require authentication, but accept any known user
071      @HTTPMethodParameter(name = service.tut.pori.tasks.Definitions.PARAMETER_BACKEND_ID) LongParameter backendId,
072      @HTTPMethodParameter(name = service.tut.pori.tasks.Definitions.PARAMETER_TASK_ID) StringParameter taskId,
073      @HTTPMethodParameter(name = DataGroups.PARAMETER_DEFAULT_NAME, required = false) DataGroups dataGroups,
074      @HTTPMethodParameter(name = Limits.PARAMETER_DEFAULT_NAME, required = false) Limits limits
075      ) 
076  {
077    return new Response(FuzzyFrontendCore.queryTaskDetails(authenticatedUser.getUserIdentity(), backendId.getValue(), taskId.getValue(), dataGroups, limits));
078  }
079  
080  /**
081   * Implementation of create task for fuzzy sensors
082   * 
083   * @param authenticatedUser 
084   * @param xml Only the workload data should be in the body. See {@link service.tut.pori.fuzzysensors.FuzzyTask}
085   * @return {@link service.tut.pori.tasks.datatypes.TaskList} with identifier for the generated task
086   */
087  @HTTPServiceMethod(name = service.tut.pori.apilta.sensors.Definitions.METHOD_CREATE_TASK, acceptedMethods={core.tut.pori.http.Definitions.METHOD_POST})
088  public Response createTask (
089      @HTTPAuthenticationParameter AuthenticationParameter authenticatedUser, // require authentication, but accept any known user
090      @HTTPMethodParameter(name = InputStreamParameter.PARAMETER_DEFAULT_NAME, bodyParameter = true) InputStreamParameter xml
091      )
092  {
093    String taskId = FuzzyFrontendCore.createTask(authenticatedUser.getUserIdentity(), _formatter.toObject(xml.getValue(), FuzzyTask.class));
094    TaskList taskList = new TaskList();
095    Task task = new SensorTask();
096    task.addTaskId(taskId);
097    taskList.addTask(task);
098    return new Response(taskList);
099  }
100}
