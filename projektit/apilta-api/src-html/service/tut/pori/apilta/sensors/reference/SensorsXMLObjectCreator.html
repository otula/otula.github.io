057public class SensorsXMLObjectCreator {
058  private static final Logger LOGGER = Logger.getLogger(SensorsXMLObjectCreator.class);
059  private static final int TEXT_LENGTH = 64;
060  private Random _random = null;
061  private RandomStringGenerator _stringGenerator = null;
062  
063  /**
064   * 
065   * @param seed for random generator, or null to use default (system time in nanoseconds)
066   */
067  public SensorsXMLObjectCreator(Long seed){
068    if(seed == null){
069      seed = System.nanoTime();
070    }
071    _random = new Random(seed);
072    _stringGenerator = new RandomStringGenerator.Builder().withinRange('a', 'z').build();
073  }
074
075  /**
076   * @return the random
077   */
078  public Random getRandom() {
079    return _random;
080  }

082  /**
083   * 
084   * @param backendIdFilter 
085   * @param createdFilter 
086   * @param dataGroups 
087   * @param limits
088   * @param measurementIdFilter 
089   * @return list of measurement objects
090   */
091  public List<Measurement> generateMeasurementList(long[] backendIdFilter, Set<Interval> createdFilter, DataGroups dataGroups, Limits limits, List<String> measurementIdFilter){
092    int count = limits.getMaxItems(service.tut.pori.apilta.sensors.datatypes.Definitions.ELEMENT_MEASUREMENT_LIST);
093    if(count < 1){
094      LOGGER.warn("count < 1");
095      return null;
096    }else if(count >= Limits.DEFAULT_MAX_ITEMS){
097      LOGGER.debug("Count was "+Limits.DEFAULT_MAX_ITEMS+", using 1.");
098      count = 1;
099    }
100    boolean hasMeasurementIdFilter = (measurementIdFilter != null && !measurementIdFilter.isEmpty());
101    if(hasMeasurementIdFilter){
102      int filterSize = measurementIdFilter.size();
103      if(hasMeasurementIdFilter && filterSize > count){
104        LOGGER.debug("Not enough measurement ids in the filter for limit count, restricting count to: "+filterSize);
105        count = filterSize;
106      }
107    }
108    
109    List<Measurement> list = new ArrayList<>();
110    for(int i=0;i<count;++i) {
111      Measurement measurement = generateMeasurementData(backendIdFilter, createdFilter, dataGroups, limits);
112      if(hasMeasurementIdFilter){ // if filter given ...
113        measurement.setMeasurementId(measurementIdFilter.get(i)); // ... modify ids to match the filter
114      }
115      list.add(measurement);
116    }
117    return (list.isEmpty() ? null : list);
118  }

120  /**
121   * 
122   * @param backendIdFilter 
123   * @param createdFilter 
124   * @param dataGroups 
125   * @param limits
126   * @return pseudo randomly generated measurement data
127   */
128  public Measurement generateMeasurementData(long[] backendIdFilter, Set<Interval> createdFilter, DataGroups dataGroups, Limits limits) {
129    Measurement data = new Measurement();
130    data.setBackendId((ArrayUtils.isEmpty(backendIdFilter) ? Math.abs(_random.nextLong()) : backendIdFilter[_random.nextInt(backendIdFilter.length)]));
131    if(DataGroups.hasDataGroup(DataGroups.DATA_GROUP_ALL, dataGroups) || DataGroups.hasDataGroup(Definitions.DATA_GROUP_DATA_POINTS, dataGroups)){
132      data.setDataPoints(createDataPointList(data.getMeasurementId(), limits, createdFilter));
133    }
134    data.setMeasurementId(UUID.randomUUID().toString());
135
136    return data;
137  }

155  /**
156   * 
157   * @param backendId 
158   * @param dataGroups
159   * @param limits
160   * @param taskType
161   * @return pseudo randomly generated task results
162   * @throws IllegalArgumentException on invalid arguments
163   */
164  public SensorTask generateTaskResults(Long backendId, DataGroups dataGroups, Limits limits, String taskType) throws IllegalArgumentException
165  {
166    SensorTask task = (SensorTask) setTaskDetails(new SensorTask(), _random.nextLong(), taskType);
167    //erase a few non-needed members for this case
168    List<TaskBackend> backends = task.getBackends();
169    backends.clear();
170    TaskBackend tb = generateTaskBackend(backendId);
171    backends.add(tb);
172    task.setCreated(null);
173    task.setUpdated(null);
174    task.setUserId(null);
175    task.setName(null);
176    task.setDescription(null);
177    task.setState(null);
178    //generate & set task results
179    MeasurementList data = new MeasurementList();
180    data.setMeasurements(generateMeasurementList((backendId == null ? new long[]{tb.getBackendId()} : new long[]{backendId}), null, dataGroups, limits, null));
181    task.setMeasurements(data);
182    
183    return task;
184  }

187   * 
188   * @param measurementId
189   * @param limits
190   * @param createdFilter 
191   * @return list of datapoint objects
192   */
193  public List<DataPoint> createDataPointList(String measurementId, Limits limits, Set<Interval> createdFilter){
194    int count = limits.getMaxItems(service.tut.pori.apilta.sensors.datatypes.Definitions.ELEMENT_DATAPOINT_LIST);
195    if(count < 1){
196      LOGGER.warn("count < 1");
197      return null;
198    }else if(count >= Limits.DEFAULT_MAX_ITEMS){
199      LOGGER.debug("Count was "+Limits.DEFAULT_MAX_ITEMS+", using 1.");
200      count = 1;
201    }
202    List<DataPoint> list = new ArrayList<>();
203    for(int i=0;i<count;++i){
204      DataPoint kw = createDataPoint(createdFilter, measurementId);
205      if(kw != null){
206        list.add(kw);
207      }
208    }
209    return (list.isEmpty() ? null : list);
210  }

212  /**
213   * Return pseudo randomly generated DataPoint
214   * @param createdFilter 
215   * @param measurementId
216   * @return a datapoint
217   */
218  public DataPoint createDataPoint(Set<Interval> createdFilter, String measurementId){
219    DataPoint dp = new DataPoint();
220    Date created = createDate(createdFilter, null);
221    dp.setCreated(created);
222    dp.setDataPointId(UUID.randomUUID().toString());
223    
224    if(measurementId == null){
225      measurementId = UUID.randomUUID().toString();
226    }
227    
228    dp.setMeasurementId(measurementId);
229    dp.setDescription(_stringGenerator.generate(TEXT_LENGTH));
230    dp.setKey(_stringGenerator.generate(TEXT_LENGTH));
231    dp.setValue(_stringGenerator.generate(TEXT_LENGTH));
232    
233    return dp;
234  }

236  /**
237   * 
238   * @param intervals if null, the returned date will be random, if given an interval will be randomly selected amongst the given intervals and a new date will be generated that is within the interval
239   * @param start override the start time
240   * @return new date created by the given intervals
241   * @throws IllegalArgumentException on invalid interval and/or start time
242   */
243  private Date createDate(Set<Interval> intervals, Date start) throws IllegalArgumentException{
244    if(intervals != null && !intervals.isEmpty()){
245      Interval interval = IterableUtils.get(intervals, _random.nextInt(intervals.size()));
246      if(start == null){
247        start = interval.getStart();
248      }
249      Date end = interval.getEnd();
250      if(end.before(start)){
251        throw new IllegalArgumentException("Cannot create valid date based on the given start time.");
252      }
253      
254      return new Date(RandomUtils.nextLong(start.getTime(), end.getTime()));
255    }else if(start == null){
256      return new Date(RandomUtils.nextLong(0, System.currentTimeMillis()));
257    }else{
258      return new Date(RandomUtils.nextLong(start.getTime(), System.currentTimeMillis()));
259    }
260  }

262  /**
263   * 
264   * @param task
265   * @param backendId
266   * @param taskType
267   * @return the populated base class task
268   */
269  public Task setTaskDetails(Task task, long backendId, String taskType){
270    if(task == null){
271      return null;
272    }
273    
274    ArrayList<TaskBackend> backends = new ArrayList<>(1);
275    TaskBackend backend = new TaskBackend();
276    backend.setBackendId(backendId);
277    backends.add(backend);
278    task.setBackends(backends);
279    
280    task.setCreated(new Date(System.currentTimeMillis() - Math.abs(_random.nextLong() % 31536000000L)));
281    task.addTaskId(UUID.randomUUID().toString());
282    HashSet<String> taskTypes = new HashSet<>(1);
283    if(StringUtils.isEmpty(taskType)){
284      taskTypes.add(_stringGenerator.generate(TEXT_LENGTH));
285    }else{
286      taskTypes.add(taskType);
287    }
288    task.setTaskTypes(taskTypes);
289    task.setUpdated(new Date(task.getCreated().getTime() + Math.abs(_random.nextLong() % 31536000000L)));
290    task.setUserId(new UserIdentity(Math.abs(_random.nextLong())));
291    task.setName(_stringGenerator.generate(TEXT_LENGTH));
292    task.setDescription(_stringGenerator.generate(TEXT_LENGTH));
293    task.setDataVisibility(generateVisibility());
294    task.setState(generateState());
295    
296    return task;
297  }

323    TaskBackend tb = new TaskBackend();

332   * @param limits 
333   * @param task
334   * @return return populated sensor task
335   */
336  public Task setSensorTaskDetails(Limits limits, SensorTask task){
337    List<Output> outputs = new ArrayList<>();
338    for(int i=0; i<_random.nextInt(4)+1; ++i){
339      Output output = new Output();
340      output.setFeature(_stringGenerator.generate(5));
341      outputs.add(output);
342    }
343    task.setOutput(outputs);
344    
345    task.setConditions(createConditionList(limits));
346    return task;
347  }

349  /**
350   * 
351   * @param limits
352   * @return list of generated random conditions
353   */
354  public List<Condition> createConditionList(Limits limits){
355    int count = limits.getMaxItems(service.tut.pori.apilta.sensors.datatypes.Definitions.ELEMENT_WHEN);
356    if(count < 1){
357      LOGGER.warn("count < 1");
358      return null;
359    }else if(count >= Limits.DEFAULT_MAX_ITEMS){
360      LOGGER.debug("Count was "+Limits.DEFAULT_MAX_ITEMS+", using 1.");
361      count = 1;
362    }
363    List<Condition> list = new ArrayList<>();
364    for(int i=0;i<count;++i){
365      Condition condition = createCondition(limits);
366      if(condition != null){
367        list.add(condition);
368      }
369    }
370    return (list.isEmpty() ? null : list);
371  }

373  /**
374   * 
375   * @param limits 
376   * @return return randomized condition
377   */
378  public Condition createCondition(Limits limits){
379    int count = limits.getMaxItems(service.tut.pori.apilta.sensors.datatypes.Definitions.ELEMENT_CONDITION);
380    if(count < 1){
381      LOGGER.warn("count < 1");
382      return null;
383    }else if(count >= Limits.DEFAULT_MAX_ITEMS){
384      LOGGER.debug("Count was "+Limits.DEFAULT_MAX_ITEMS+", using 1.");
385      count = 1;
386    }
387    Condition condition = new Condition();
388    TreeMap<String, String> conditions = new TreeMap<>();
389    for(int i=0; i<_random.nextInt(2)+1; ++i){
390      conditions.put(_stringGenerator.generate(5), _stringGenerator.generate(10));
391    }
392    conditions.put("time/validFromToRange", Instant.now().minusSeconds(Math.abs(_random.nextLong() % 31536000L)).toString() +"/"+Instant.now().plusSeconds(Math.abs(_random.nextLong() % 31536000L)).toString());
393    condition.setConditions(conditions);
394    return condition;
395  }
