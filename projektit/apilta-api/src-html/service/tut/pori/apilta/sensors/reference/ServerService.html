039/**
040 * 
041 * Reference implementation for Server APIs.
042 * 
043 * <h1>Implementation Service path {@value service.tut.pori.apilta.sensors.Definitions#SERVICE_SENSORS}</h1>
044 * 
045 * @see service.tut.pori.apilta.sensors.SensorService
046 *
047 */
048@HTTPService(name = Definitions.SERVICE_SENSORS_REFERENCE_SERVER)
049public class ServerService {
050  private static final Logger LOGGER = Logger.getLogger(ServerService.class);
051  private XMLFormatter _formatter = new XMLFormatter();
052  
053  /**
054   * The request is to be sent in the body of POST method. The Content-Type header MUST be set to "text/xml". The character set MUST be UTF-8. For example, "Content-Type: text/xml; charset=UTF-8".
055   * 
056   * The method can be called multiple times by the back-ends. Each call is assumed to be an incremental update on the previous one and can contain any amount of new information. 
057   * 
058   * <h2>Example Query:</h2>
059   *
060   * POST /rest/{@value service.tut.pori.apilta.sensors.reference.Definitions#SERVICE_SENSORS_REFERENCE_SERVER}/{@value service.tut.pori.tasks.Definitions#METHOD_TASK_FINISHED}<br>
061   * Content-Type: text/xml; charset=UTF-8<br><br>
062   *
063   * <b>[HTTP BODY STARTS]</b><br>
064   * 
065   * {@doc.restlet service="[service.tut.pori.apilta.sensors.reference.Definitions#SERVICE_SENSORS_REFERENCE_EXAMPLE]" method="[service.tut.pori.apilta.sensors.reference.Definitions#METHOD_TASK_RESULTS]" type="GET" query="" body_uri=""} <br>
066   * 
067   * <b>[HTTP BODY ENDS]</b><br>
068   *
069   * <h2>Example Result:</h2>
070   * 
071   * {@doc.restlet service="[service.tut.pori.apilta.sensors.reference.Definitions#SERVICE_SENSORS_REFERENCE_SERVER]" method="[service.tut.pori.tasks.Definitions#METHOD_TASK_FINISHED]" type="POST" query="" body_uri="[service.tut.pori.apilta.sensors.reference.Definitions#SERVICE_SENSORS_REFERENCE_EXAMPLE]/[service.tut.pori.apilta.sensors.reference.Definitions#METHOD_TASK_RESULTS]?[core.tut.pori.http.parameters.DataGroups#PARAMETER_DEFAULT_NAME]=[service.tut.pori.apilta.sensors.Definitions#DATA_GROUP_DATA_POINTS]"}
072   * 
073   * @param authenticatedUser 
074   * @param xml Only the result data should be in the body. See {@link service.tut.pori.apilta.sensors.datatypes.SensorTask}
075   */
076  @HTTPServiceMethod(name = service.tut.pori.tasks.Definitions.METHOD_TASK_FINISHED, acceptedMethods={core.tut.pori.http.Definitions.METHOD_POST})
077  public void taskFinished (
078      @HTTPAuthenticationParameter(required = false) AuthenticationParameter authenticatedUser,
079      @HTTPMethodParameter(name = InputStreamParameter.PARAMETER_DEFAULT_NAME, bodyParameter = true) InputStreamParameter xml
080      ) 
081  {
082    try {
083      String body = IOUtils.toString(xml.getValue(), core.tut.pori.http.Definitions.CHARSET_UTF8); // read the body
084      LOGGER.debug(body); // print to debug
085      try(InputStream input = IOUtils.toInputStream(body, core.tut.pori.http.Definitions.CHARSET_UTF8)){ // convert back to stream for unmarshal
086        SensorsReferenceCore.taskFinished(authenticatedUser.getUserIdentity(), _formatter.toObject(input, SensorTask.class));
087      }
088    } catch (IOException ex) {
089      LOGGER.error(ex, ex);
090    }
091  }
092  
093  /**
094   * This method can be used to retrieve the up-to-date details and progress of a previously scheduled task.
095   * 
096   * <h2>Example Query:</h2>
097   *
098   * GET /rest/{@value service.tut.pori.apilta.sensors.reference.Definitions#SERVICE_SENSORS_REFERENCE_SERVER}/{@value service.tut.pori.tasks.Definitions#METHOD_QUERY_TASK_DETAILS}?{@value service.tut.pori.tasks.Definitions#PARAMETER_BACKEND_ID}=1&amp;{@value service.tut.pori.tasks.Definitions#PARAMETER_TASK_ID}=1<br>
099   *
100   * <h2>Example Result:</h2>
101   * 
102   * {@doc.restlet service="[service.tut.pori.apilta.sensors.reference.Definitions#SERVICE_SENSORS_REFERENCE_SERVER]" method="[service.tut.pori.tasks.Definitions#METHOD_QUERY_TASK_DETAILS]" type="GET" query="[service.tut.pori.tasks.Definitions#PARAMETER_BACKEND_ID]=1&[service.tut.pori.tasks.Definitions#PARAMETER_TASK_ID]=1" body_uri=""}
103   * 
104   * @param authenticatedUser 
105   * @param taskId
106   * @param backendId
107   * @param dataGroups
108   * @param limits paging limits
109   * @return response See {@link service.tut.pori.apilta.sensors.datatypes.SensorTask}
110   */
111  @HTTPServiceMethod(name = service.tut.pori.tasks.Definitions.METHOD_QUERY_TASK_DETAILS, acceptedMethods={core.tut.pori.http.Definitions.METHOD_GET})
112  public Response queryTaskDetails(
113      @HTTPAuthenticationParameter(required = false) AuthenticationParameter authenticatedUser,
114      @HTTPMethodParameter(name = service.tut.pori.tasks.Definitions.PARAMETER_BACKEND_ID) LongParameter backendId,
115      @HTTPMethodParameter(name = service.tut.pori.tasks.Definitions.PARAMETER_TASK_ID) LongParameter taskId,
116      @HTTPMethodParameter(name = DataGroups.PARAMETER_DEFAULT_NAME, required = false) DataGroups dataGroups,
117      @HTTPMethodParameter(name = Limits.PARAMETER_DEFAULT_NAME, required = false) Limits limits
118      ) 
119  {
120    return new Response(SensorsReferenceCore.generateTaskDetails(authenticatedUser.getUserIdentity(), backendId.getValue(), dataGroups, limits, service.tut.pori.apilta.sensors.Definitions.TASK_TYPE_DATA_COLLECT)); // we can simply return pseudo randomly generated task details with hard-coded task type
121  }
122  
123  /**
124   * The request is to be sent in the body of POST method.
125   * 
126   * <h2>Example Query:</h2>
127   *
128   * POST /rest/{@value service.tut.pori.apilta.sensors.reference.Definitions#SERVICE_SENSORS_REFERENCE_SERVER}/{@value service.tut.pori.apilta.sensors.Definitions#METHOD_CREATE_FILE}?{@value service.tut.pori.tasks.Definitions#PARAMETER_BACKEND_ID}=1<br>
129   * Content-Type: text/xml; charset=UTF-8<br><br>
130   *
131   * <b>[HTTP BODY STARTS]</b><br>
132   * 
133   * [ANY FILE CONTENT] <br>
134   * 
135   * <b>[HTTP BODY ENDS]</b><br>
136   *
137   * <h2>Example Result:</h2>
138   * 
139   * {@doc.restlet service="[service.tut.pori.apilta.sensors.reference.Definitions#SERVICE_SENSORS_REFERENCE_SERVER]" method="[service.tut.pori.apilta.sensors.Definitions#METHOD_CREATE_FILE]" type="POST" query="[service.tut.pori.tasks.Definitions#PARAMETER_BACKEND_ID]=1" body_uri="[service.tut.pori.apilta.sensors.reference.Definitions#SERVICE_SENSORS_REFERENCE_EXAMPLE]/[service.tut.pori.tasks.Definitions#ELEMENT_TASK]"}
140   * 
141   * @param authenticatedUser 
142   * @param backendId 
143   * @param file only the file contents should be in the body.
144   * @return {@link service.tut.pori.apilta.files.datatypes.FileDetailsList} with identifier for the generated file
145   */
146  @HTTPServiceMethod(name = service.tut.pori.apilta.sensors.Definitions.METHOD_CREATE_FILE, acceptedMethods={core.tut.pori.http.Definitions.METHOD_POST})
147  public Response createFile (
148      @HTTPAuthenticationParameter(required = false) AuthenticationParameter authenticatedUser,
149      @HTTPMethodParameter(name = service.tut.pori.tasks.Definitions.PARAMETER_BACKEND_ID) LongParameter backendId,
150      @HTTPMethodParameter(name = InputStreamParameter.PARAMETER_DEFAULT_NAME, bodyParameter = true) InputStreamParameter file
151      ) 
152  {
153    FileDetailsList list = new FileDetailsList();
154    FileDetails details = new FileDetails();
155    details.setGUID(SensorsReferenceCore.createFile(authenticatedUser.getUserIdentity(), backendId.getValue(), file.getValue()));
156    list.addFile(details);
157    return new Response(list);
158  }
